'LOAD FROM DATA FILE INTO AN ARRAY THE NAMES OF THE MONTHS THEN
'OUTPUT THEM USING A FOR NEXT LOOP

'MAINLINE:

    GOSUB[HOUSEKEEPING]
    GOSUB[PROCESS]
    GOSUB[WRAPUP]
end

[HOUSEKEEPING]
      GOSUB [INITIALIZE]
      GOSUB [OPENFILE]
      GOSUB [LOADARRAYDATA]
RETURN

[PROCESS]
      GOSUB [OUTPUTARRAYDATA]
      GOSUB [OUTPUTTODAYLONGDATE]
RETURN

[WRAPUP]
      GOSUB [CLOSEFILE]
RETURN

[INITIALIZE]
      DIM MONTHNAMES$(12)
RETURN

[OPENFILE]
    OPEN "MonthNames.DAT" for input as #MonthNamesFile
RETURN

[LOADARRAYDATA]
      FOR X = 1 TO 12
        INPUT #MonthNamesFile, MONTHNAMES$(X)
      NEXT
RETURN

[OUTPUTARRAYDATA]
    FOR X = 1 TO 12
        PRINT MONTHNAMES$(X)
    NEXT
    PRINT
RETURN

[OUTPUTTODAYLONGDATE]
      MONTH = 12
      DAY = 7
      YEAR = 2009

      PRINT MONTHNAMES$(MONTH); " "; DAY; ", "; YEAR
      PRINT MONTH; " "; DAY; ", "; YEAR

      '2ND DATE

      MONTH = 2
      DAY = 14
      YEAR = 2010
      PRINT MONTHNAMES$(MONTH); " "; DAY; ", "; YEAR

      '3RD DATE
      MONTH = 3
      DAY = 17
      YEAR = 2012
      PRINT MONTHNAMES$(MONTH); " "; DAY; ", "; YEAR
RETURN

[CLOSEFILE]
      CLOSE #MonthNamesFile
RETURN
